def groupAnagrams(words):
    result = []

    for i in range(len(words)):
        found = False
        for j in range(len(result)):
            a = list(words[i])
            b = list(result[j][0])
            a.sort()
            b.sort()
            if a == b:
                result[j].append(words[i])
                found = True
                break

        if not found:
            result.append([words[i]])

    return result


words = ["eat", "tea", "tan", "ate", "nat", "bat"]
print(groupAnagrams(words))

Expected Output:

[['eat', 'tea', 'ate'], ['tan', 'nat'], ['bat']]
